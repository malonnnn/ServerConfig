ma:/ma = >>>

@magnitude = 2; // speed of falling block
@damage = 50; // amount of damage the block will do to other players

// get the starting location
@loc = ploc();
@loc['y'] += 2; // chest height

// get the vector from player yaw/pitch
@yaw = @loc['yaw'] + 90;
@pitch = neg(@loc['pitch']);
@vector = associative_array();
@vector['x'] = @magnitude * (cos(to_radians(@yaw)) * cos(to_radians(@pitch)));
@vector['y'] = @magnitude * sin(to_radians(@pitch));
@vector['z'] = @magnitude * (sin(to_radians(@yaw)) * cos(to_radians(@pitch)));

@id = spawn_falling_block(@loc, 42, @vector);

bind(entity_change_block, associative_array('id': 'magneto'.@id), associative_array('to': 42), 
        @event, @id, @damage, @player = player()) {
            
            @loc = entity_loc(@id);
            drop_item(@loc,array(type: 42, qty: 1))
            cancel()
            
}

set_interval(50, closure(){
    try {
        @loc = entity_loc(@id);
        @loc['y'] -= 1; // so we detect if the block is at player chest height
        foreach(@p in entities_in_radius(@loc, 3)) {
            // apply damage to player's UUID and attribute it to current player
            if(is_entity_living(@p)){
                damage_entity(puuid(@p), @damage, puuid());
            }
        }
    } catch(Exception @ex) {
        // entity no longer exists, clear the interval
        clear_task();
        // unbind the landing event
        unbind('magneto'.@id);
    }});

<<<

    @sourcePlayer = player()
    
    # let's toggle based on their mode
    if(pmode() == 'SPECTATOR') {
        @activeTarget = pspectator_target(@sourcePlayer)
        if(@activeTarget)
        {
            tmsg(@activeTarget,concat(color(dark_gray), @sourcePlayer 'is no longer spectating you.'))
        }
    
        # teleports them back to spawn, but you can change this
        set_ploc(get_spawn());
        # if the default mode is ADVENTURE, you can change this too
        set_pmode('SURVIVAL');
    } else {
        if($player) {
            @targetPlayer = $player
            if (pworld(@targetPlayer) == pworld(@sourcePlayer)) {
                if(puuid(@targetPlayer) != puuid(@sourcePlayer)) {
                    set_pmode('SPECTATOR');
                    set_pspectator_target(puuid(@targetPlayer));
                    tmsg(@targetPlayer,concat(color(dark_gray), @sourcePlayer 'is spectating you.'))
                } else {
                    tmsg(@sourcePlayer,concat(color(red), 'You are already spec\'ing yourself, dumb-dumb.'))
                }
            } else {
                tmsg(@sourcePlayer,concat(color(red), 'You can only spec people in the same world as you.'))
            }
        } else {
            # they didn't specify a player, so let's go to the first player in the world.
            # this could also be changed, depending on the context.
            # maybe you want to grab a player from a game's scoreboard or a custom list
            @plrs = all_players(pworld())
            if(array_size(@plrs) > 1) {
                if(puuid(@sourcePlayer) != puuid(@plrs[0])) {
                    set_pmode('SPECTATOR');
                    set_pspectator_target(puuid(@plrs[0]));
                    tmsg(@plrs[0],concat(color(dark_gray), @sourcePlayer 'is spectating you.'))
                } else {
                    set_pmode('SPECTATOR');
                    set_pspectator_target(puuid(@plrs[1]));
                    tmsg(@plrs[1],concat(color(dark_gray), @sourcePlayer 'is spectating you.'))
                }
            } else {
                tmsg(@sourcePlayer,concat(color(red), 'No one in this world to spec.'))
            }
        }
    }

/resetskywars $plr = >>>

    # How many seconds they have to wait before being able
    # to re-run fixspleef
    assign(@cooldown, 300)

    # Ensure the persistent variable exists. If it doesn't, set it to 0.
    if(is_null(get_value('skywars_lastreset')),
        store_value('skywars_lastreset',0)
    )
    
    # Get our persistent value
    assign(@lastreset, get_value('skywars_lastreset'))

    # Note that time is in msec, not sec. So we always divide it by 1000.
    if(@lastreset + @cooldown > time() / 1000){
        # then
        assign(@timeleft, subtract(add(@lastreset, @cooldown), divide(time(),1000)))
        tmsg($plr,concat('§c[§4', Error, '§c] §7You must wait another §6', round(@timeleft), ' §7seconds before you can reset §4Skywars §7again!'))
        die()
    }

    # Save the current time() (in seconds) in spleef1_lastreset
    store_value('skywars_lastreset', time() / 1000)

    shell_adv('/home/minecraft/resetskywars.sh',
    array(subshell: true,
        workingDir: '/home/minecraft/'
        )
